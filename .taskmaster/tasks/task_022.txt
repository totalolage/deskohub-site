2. **Create Contact Form Component**:
   ```typescript
   // components/contact/contact-form.tsx
   'use client';
   
   import { useState } from 'react';
   import { createTranslator } from '@/paraglide/runtime';
   import { Button } from '@/components/ui/button';
   import { Input } from '@/components/ui/input';
   import { Textarea } from '@/components/ui/textarea';
   import { sendContactMessage } from '@/app/actions';
   
   export function ContactForm({ locale }) {
     const t = createTranslator(locale);
     const [formState, setFormState] = useState({
       name: '',
       email: '',
       phone: '',
       subject: '',
       message: '',
     });
     const [errors, setErrors] = useState({});
     const [isSubmitting, setIsSubmitting] = useState(false);
     const [submitStatus, setSubmitStatus] = useState(null);
     
     const handleChange = (e) => {
       const { name, value } = e.target;
       setFormState(prev => ({ ...prev, [name]: value }));
     };
     
     const validateForm = () => {
       const newErrors = {};
       if (!formState.name) newErrors.name = t('contact.form.errors.nameRequired');
       if (!formState.email) newErrors.email = t('contact.form.errors.emailRequired');
       else if (!/\S+@\S+\.\S+/.test(formState.email)) newErrors.email = t('contact.form.errors.emailInvalid');
       if (!formState.message) newErrors.message = t('contact.form.errors.messageRequired');
       
       setErrors(newErrors);
       return Object.keys(newErrors).length === 0;
     };
     
     const handleSubmit = async (e) => {
       e.preventDefault();
       if (!validateForm()) return;
       
       setIsSubmitting(true);
       try {
         const result = await sendContactMessage(formState);
         if (result.success) {
           setSubmitStatus('success');
           setFormState({
             name: '',
             email: '',
             phone: '',
             subject: '',
             message: '',
           });
         } else {
           setSubmitStatus('error');
         }
       } catch (error) {
         setSubmitStatus('error');
       } finally {
         setIsSubmitting(false);
       }
     };
     
     return (
       <div className="bg-white dark:bg-gray-800 p-6 rounded-lg shadow-md">
         <h2 className="text-2xl font-bold mb-6">{t('contact.form.title')}</h2>
         
         {submitStatus === 'success' && (
           <div className="bg-green-100 dark:bg-green-900 text-green-800 dark:text-green-100 p-4 mb-6 rounded">
             {t('contact.form.successMessage')}
           </div>
         )}
         
         {submitStatus === 'error' && (
           <div className="bg-red-100 dark:bg-red-900 text-red-800 dark:text-red-100 p-4 mb-6 rounded">
             {t('contact.form.errorMessage')}
           </div>
         )}
         
         <form onSubmit={handleSubmit}>
           <div className="mb-4">
             <label htmlFor="name" className="block mb-2">{t('contact.form.nameLabel')}</label>
             <Input
               id="name"
               name="name"
               value={formState.name}
               onChange={handleChange}
               error={errors.name}
             />
             {errors.name && <p className="text-red-500 mt-1 text-sm">{errors.name}</p>}
           </div>
           
           <div className="mb-4">
             <label htmlFor="email" className="block mb-2">{t('contact.form.emailLabel')}</label>
             <Input
               id="email"
               name="email"
               type="email"
               value={formState.email}
               onChange={handleChange}
               error={errors.email}
             />
             {errors.email && <p className="text-red-500 mt-1 text-sm">{errors.email}</p>}
           </div>
           
           <div className="mb-4">
             <label htmlFor="phone" className="block mb-2">{t('contact.form.phoneLabel')}</label>
             <Input
               id="phone"
               name="phone"
               type="tel"
               value={formState.phone}
               onChange={handleChange}
             />
           </div>
           
           <div className="mb-4">
             <label htmlFor="subject" className="block mb-2">{t('contact.form.subjectLabel')}</label>
             <Input
               id="subject"
               name="subject"
               value={formState.subject}
               onChange={handleChange}
             />
           </div>
           
           <div className="mb-6">
             <label htmlFor="message" className="block mb-2">{t('contact.form.messageLabel')}</label>
             <Textarea
               id="message"
               name="message"
               rows={5}
               value={formState.message}
               onChange={handleChange}
               error={errors.message}
             />
             {errors.message && <p className="text-red-500 mt-1 text-sm">{errors.message}</p>}
           </div>
           
           <Button 
             type="submit" 
             disabled={isSubmitting}
             className="w-full"
           >
             {isSubmitting ? t('contact.form.submitting') : t('contact.form.submit')}
           </Button>
         </form>
       </div>
     );
   }
   ```

   import { createTranslator } from '@/paraglide/runtime';
   import { cookies } from 'next/headers';
   
   export async function sendContactMessage(formData) {
     // Get the current locale from cookies
     const cookieStore = cookies();
     const locale = cookieStore.get('NEXT_LOCALE')?.value || 'cs';
     const t = createTranslator(locale);
     
     try {
       // Validate the form data
       if (!formData.name || !formData.email || !formData.message) {
         return {
           success: false,
           message: t('contact.form.errors.invalidData'),
         };
       }
       
       // Here you would typically send the message to your backend
       // For example, using an API endpoint or email service
       
       // For now, we'll simulate a successful submission
       console.log('Contact form submission:', formData);
       
       // Add a slight delay to simulate network request
       await new Promise(resolve => setTimeout(resolve, 1000));
       
       return {
         success: true,
         message: t('contact.form.successMessage'),
       };
     } catch (error) {
       console.error('Error submitting contact form:', error);
       return {
         success: false,
         message: t('contact.form.errorMessage'),
       };
     }

# Test Strategy:
## Test Strategy

1. **Functionality Testing**:
   - Verify the Contact page loads correctly at `/contact` and `/en/contact` routes
   - Test that all components render properly with correct content
   - Confirm that the contact form validation works as expected:
     - Submit with empty required fields and verify error messages
     - Submit with invalid email format and verify error message
     - Submit with all valid data and verify success message
   - Test that the map loads correctly and is interactive
   - Verify that all links (social media, email, phone) are correctly formatted and clickable

2. **Internationalization Testing**:
   - Switch between Czech and English languages and verify all content is properly translated
   - Check that form validation messages appear in the correct language
   - Verify that form submission success/error messages are displayed in the current language
   - Test that business hours, team positions, and other localized content displays correctly in both languages

3. **Responsive Design Testing**:
   - Test the page layout on various screen sizes:
     - Mobile phones (320px - 480px)
     - Tablets (768px - 1024px)
     - Desktops (1024px+)
   - Verify that the layout adjusts appropriately:
     - Form and contact info stack vertically on mobile
     - Team members display in a single column on mobile, multiple columns on larger screens
     - Map is properly sized and usable on all devices

4. **Accessibility Testing**:
   - Run automated accessibility tests using tools like Lighthouse or axe
   - Verify proper heading hierarchy (h1, h2, h3)
   - Check that all form inputs have associated labels
   - Ensure all images have appropriate alt text
   - Test keyboard navigation through the form and page
   - Verify sufficient color contrast for text elements

5. **Form Submission Testing**:
   - Test the server action by submitting the form with valid data
   - Verify that the form data is correctly logged or processed
   - Test error handling by simulating server errors
   - Verify loading states display correctly during form submission

6. **SEO Testing**:
   - Verify that metadata is correctly generated for both languages
   - Check that page title and description are appropriate
   - Confirm that Open Graph tags are properly set for social sharing
   - Test that the page is properly indexed by search engines

7. **Integration Testing**:
   - Verify that the Contact page is correctly linked from the main navigation
