// components/training-rooms/filters.tsx
import { useState } from 'react';
import { createTranslator } from '@/paraglide/runtime';
import { useParams } from 'next/navigation';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Slider } from '@/components/ui/slider';
import { Checkbox } from '@/components/ui/checkbox';
import { DatePicker } from '@/components/ui/date-picker';

export function TrainingRoomFilters() {
  const { locale } = useParams();
  const t = createTranslator(locale as string);
  
  const [capacity, setCapacity] = useState([10]);
  const [date, setDate] = useState<Date | undefined>(undefined);
  const [equipment, setEquipment] = useState({
    projector: false,
    whiteboard: false,
    videoConference: false,
    highSpeedWifi: false,
  });
  
  const handleReset = () => {
    setCapacity([10]);
    setDate(undefined);
    setEquipment({
      projector: false,
      whiteboard: false,
      videoConference: false,
      highSpeedWifi: false,
    });
  };
  
  return (
    <div className="bg-muted p-6 rounded-lg mb-8">
      <h3 className="text-xl font-semibold mb-4">{t('filter_rooms')}</h3>
      
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <div>
          <Label>{t('capacity')}: {capacity[0]}+ {t('people')}</Label>
          <Slider
            defaultValue={[10]}
            max={50}
            step={5}
            onValueChange={setCapacity}
          />
        </div>
        
        <div>
          <Label>{t('date')}</Label>
          <DatePicker
            selected={date}
            onSelect={setDate}
            placeholder={t('select_date')}
          />
        </div>
        
        <div>
          <Label className="mb-2 block">{t('equipment')}</Label>
          <div className="space-y-2">
            <div className="flex items-center space-x-2">
              <Checkbox 
                id="projector" 
                checked={equipment.projector}
                onCheckedChange={(checked) => 
                  setEquipment({...equipment, projector: checked as boolean})
                }
              />
              <Label htmlFor="projector">{t('projector')}</Label>
            </div>
            <div className="flex items-center space-x-2">
              <Checkbox 
                id="whiteboard" 
                checked={equipment.whiteboard}
                onCheckedChange={(checked) => 
                  setEquipment({...equipment, whiteboard: checked as boolean})
                }
              />
              <Label htmlFor="whiteboard">{t('whiteboard')}</Label>
            </div>
            <div className="flex items-center space-x-2">
              <Checkbox 
                id="videoConference" 
                checked={equipment.videoConference}
                onCheckedChange={(checked) => 
                  setEquipment({...equipment, videoConference: checked as boolean})
                }
              />
              <Label htmlFor="videoConference">{t('video_conferencing')}</Label>
            </div>
            <div className="flex items-center space-x-2">
              <Checkbox 
                id="highSpeedWifi"
