3. Implement a language switcher component that triggers proper re-renders:
```typescript
// components/LanguageSwitcher.tsx
import { useRouter } from 'next/navigation';
import { sourceLanguageTag, runtimeLanguageTag } from '@/paraglide/runtime';
import { setLanguageTag } from '@/paraglide/runtime';

export function LanguageSwitcher() {
  const router = useRouter();
  const currentLang = runtimeLanguageTag();
  
  const toggleLanguage = () => {
    // Switch between Czech and English
    const newLang = currentLang === 'cs' ? 'en' : 'cs';
    
    // Update the language in Paraglide runtime
    setLanguageTag(newLang);
    
    // Update the cookie for server-side rendering
    document.cookie = `NEXT_LOCALE=${newLang};path=/;max-age=31536000`;
    
    // Optionally refresh the current route to ensure all components update
    // This might be needed if some components don't respond to runtime changes
    router.refresh();
    
    // Dispatch a custom event that the header can listen for
    window.dispatchEvent(new CustomEvent('languagechange'));
  };
  
  return (
    <button onClick={toggleLanguage}>
      {currentLang === 'cs' ? 'Switch to English' : 'Přepnout do češtiny'}
    </button>
  );
}
```

import { render, screen, fireEvent } from '@testing-library/react';
import { Header } from '@/components/Header';

import { vi } from 'vitest';

    refresh: vi.fn(),
    push: vi.fn()
  })

  runtimeLanguageTag: vi.fn(() => 'en'),
  setLanguageTag: vi.fn()
}));

// Mock paraglide messages
vi.mock('@/paraglide/messages', () => ({
  i: (key: string) => {
    const translations: Record<string, Record<string, string>> = {
      en: {
        'nav.home': 'Home',
        'nav.about': 'About',
        'nav.booking': 'Book a Table'
      },
      cs: {
        'nav.home': 'Domů',
        'nav.about': 'O nás',
        'nav.booking': 'Rezervovat stůl'
      }
    };
    
    const lang = runtimeLanguageTag();
    return translations[lang][key] || key;
  }
}));

describe('Header Component', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });
  
  test('displays navigation in English by default', () => {
    render(<Header />);
    
    expect(screen.getByText('Home')).toBeInTheDocument();
    expect(screen.getByText('About')).toBeInTheDocument();
    expect(screen.getByText('Book a Table')).toBeInTheDocument();
  });
  
  test('updates navigation text when language changes', () => {
    render(<Header />);
    
    // Initial state - English
    expect(screen.getByText('Home')).toBeInTheDocument();
    
    // Mock language change to Czech

    // Trigger language change event
    window.dispatchEvent(new CustomEvent('languagechange'));
    
    // Should now show Czech text
    expect(screen.getByText('Domů')).toBeInTheDocument();
    expect(screen.getByText('O nás')).toBeInTheDocument();
    expect(screen.getByText('Rezervovat stůl')).toBeInTheDocument();
  });
  
  test('language switcher toggles between languages', () => {
    render(<Header />);
    
    // Find language switcher button
    const switcherButton = screen.getByText('Switch to Czech');
    
    // Click to switch to Czech
    fireEvent.click(switcherButton);
    
    // Verify setLanguageTag was called with 'cs'
    expect(setLanguageTag).toHaveBeenCalledWith('cs');
    
    // Mock the language has changed
    (runtimeLanguageTag as any).mockReturnValue('cs');
    
    // Trigger language change event
    window.dispatchEvent(new CustomEvent('languagechange'));
    
    // Button text should now be in Czech
    expect(screen.getByText('Přepnout do angličtiny')).toBeInTheDocument();
  });
