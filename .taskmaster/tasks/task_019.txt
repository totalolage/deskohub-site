2. **Create Enhanced Gallery Component**:
   ```typescript
   // components/enhanced-gallery.tsx
   import { useState } from 'react';
   import Image from 'next/image';
   import { createTranslator } from '@/paraglide/runtime';
   import { Button } from '@/components/ui/button';
   import { Dialog, DialogContent } from '@/components/ui/dialog';
   
   // Define image categories
   const CATEGORIES = ['all', 'interior', 'atmosphere', 'events', 'facilities'];
   
   // Define gallery images with metadata
   const GALLERY_IMAGES = [
     {
       src: '/images/gallery/interior-1.jpg',
       alt: 'gallery.interior.image1.alt',
       category: 'interior',
       width: 1200,
       height: 800,
     },
     // Add at least 5 images per category (20+ total)
     // ...more images
   ];
   
   export function EnhancedGallery({ locale }) {
     const t = createTranslator(locale);
     const [selectedCategory, setSelectedCategory] = useState('all');
     const [selectedImage, setSelectedImage] = useState(null);
     
     // Filter images based on selected category
     const filteredImages = selectedCategory === 'all' 
       ? GALLERY_IMAGES 
       : GALLERY_IMAGES.filter(img => img.category === selectedCategory);
     
     return (
       <div className="space-y-8">
         {/* Category Filter */}
         <div className="flex flex-wrap gap-2 justify-center mb-8">
           {CATEGORIES.map(category => (
             <Button
               key={category}
               variant={selectedCategory === category ? "default" : "outline"}
               onClick={() => setSelectedCategory(category)}
               className="capitalize"
             >
               {t(`gallery.categories.${category}`)}
             </Button>
           ))}
         </div>
         
         {/* Gallery Grid */}
         <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
           {filteredImages.map((image, index) => (
             <div 
               key={index} 
               className="relative aspect-square overflow-hidden rounded-lg cursor-pointer hover:opacity-90 transition-opacity"
               onClick={() => setSelectedImage(image)}
             >
               <Image
                 src={image.src}
                 alt={t(image.alt)}
                 fill
                 className="object-cover"
                 sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
               />
             </div>
           ))}
         </div>
         
         {/* Lightbox Dialog */}
         <Dialog open={!!selectedImage} onOpenChange={(open) => !open && setSelectedImage(null)}>
           <DialogContent className="max-w-4xl w-[90vw] max-h-[90vh] p-0 overflow-hidden">
             {selectedImage && (
               <div className="relative w-full h-full min-h-[50vh]">
                 <Image
                   src={selectedImage.src}
                   alt={t(selectedImage.alt)}
                   fill
                   className="object-contain"
                   sizes="90vw"
                 />
               </div>
             )}
           </DialogContent>
         </Dialog>
       </div>
     );
   }
   ```
