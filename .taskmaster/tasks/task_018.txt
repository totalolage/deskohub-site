   ```typescript
   // components/games-gallery.tsx
   import Image from 'next/image';
   import { createTranslator } from '@/paraglide/runtime';
   import { GameCard } from '@/components/ui/game-card';
   
   // Define game data type
   type Game = {
     id: string;
     imageUrl: string;
     playerCount: string;
     playTime: string;
     difficulty: 'easy' | 'medium' | 'hard';
   };
   
   // This could be moved to a data file or fetched from an API
   const games: Record<string, Game[]> = {
     'cs': [
       // Czech game data
       {
         id: 'catan',
         imageUrl: '/images/games/catan.jpg',
         playerCount: '3-4',
         playTime: '60-120 min',
         difficulty: 'medium',
       },
       // Add more games...
     ],
     'en': [
       // English game data
       {
         id: 'catan',
         imageUrl: '/images/games/catan.jpg',
         playerCount: '3-4',
         playTime: '60-120 min',
         difficulty: 'medium',
       },
       // Add more games...
     ]
   };
   
   export function GamesGallery({ locale }: { locale: string }) {
     const t = createTranslator(locale);
     const localizedGames = games[locale] || games['en']; // Fallback to English
     
     return (
       <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
         {localizedGames.map((game) => (
           <GameCard 
             key={game.id}
             game={game}
             title={t(`games.${game.id}.title`)}
             description={t(`games.${game.id}.description`)}
             rules={t(`games.${game.id}.rules`)}
           />
         ))}
       </div>
     );
   }
   ```

3. **Create Game Card Component**:
   ```typescript
   // components/ui/game-card.tsx
   import Image from 'next/image';
   import { useState } from 'react';
   import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogTrigger } from '@/components/ui/dialog';
   
   type GameCardProps = {
     game: {
       id: string;
       imageUrl: string;
       playerCount: string;
       playTime: string;
       difficulty: 'easy' | 'medium' | 'hard';
     };
     title: string;
     description: string;
     rules: string;
   };
   
   export function GameCard({ game, title, description, rules }: GameCardProps) {
     return (
       <div className="rounded-lg overflow-hidden shadow-md hover:shadow-lg transition-shadow">
         <div className="relative h-48">
           <Image 
             src={game.imageUrl} 
             alt={title} 
             fill 
             className="object-cover" 
           />
         </div>
         
         <div className="p-4">
           <h3 className="text-xl font-bold mb-2">{title}</h3>
           <p className="text-gray-700 dark:text-gray-300 mb-4">{description}</p>
           
           <div className="flex justify-between mb-4">
             <span className="text-sm">
               <span className="font-medium">Players:</span> {game.playerCount}
             </span>
             <span className="text-sm">
               <span className="font-medium">Time:</span> {game.playTime}
             </span>
           </div>
           
           <Dialog>
             <DialogTrigger asChild>
               <button className="btn btn-secondary w-full">View Rules</button>
             </DialogTrigger>
             <DialogContent>
               <DialogHeader>
                 <DialogTitle>{title} - Rules</DialogTitle>
               </DialogHeader>
               <div className="prose max-w-none">
                 <div dangerouslySetInnerHTML={{ __html: rules }} />
               </div>
             </DialogContent>
           </Dialog>
         </div>
       </div>
     );
   }
   ```

# Test Strategy:
## Test Strategy

1. **Functionality Testing**:
   - Verify the Board Games page loads correctly at `/board-games` and `/en/board-games` routes
   - Confirm all game cards display properly with correct images, titles, and descriptions
   - Test the "View Rules" dialog for each game to ensure it opens and displays content correctly
   - Verify the "Book a Table" button links to the correct reservation page with proper locale handling
   - Test that the page is accessible via the navigation menu

2. **Internationalization Testing**:
   - Verify all text content is properly translated in both Czech and English
   - Test language switching between `/board-games` and `/en/board-games`
   - Confirm that game-specific content (titles, descriptions, rules) displays in the correct language
   - Verify that metadata (page title, description) changes appropriately with language

3. **Responsive Design Testing**:
   - Test the page layout on various screen sizes:
     - Mobile phones (320px - 480px width)
     - Tablets (768px - 1024px width)
     - Desktop (1024px+ width)
   - Verify the game cards grid adjusts properly from 1 column on mobile to 3 columns on desktop
   - Ensure images are responsive and maintain proper aspect ratios
   - Test the dialog component on different screen sizes

4. **SEO Testing**:
   - Verify that proper metadata is generated for both language versions
   - Check that OpenGraph tags are correctly set for social sharing
   - Validate that image alt tags are present and descriptive
   - Test that the page is properly indexed by search engines using Search Console tools

5. **Performance Testing**:
   - Run Lighthouse performance tests to ensure the page loads efficiently
   - Verify that images are properly optimized and lazy-loaded
   - Check for any render-blocking resources that might slow down page load
   - Test the time-to-interactive metric, especially with multiple game cards

6. **Accessibility Testing**:
   - Verify proper heading hierarchy (h1, h2, h3) throughout the page
   - Test keyboard navigation for all interactive elements
   - Ensure sufficient color contrast for text elements
   - Verify that the dialog component is accessible via keyboard and screen readers
   - Run automated accessibility tests using tools like Axe or Lighthouse
