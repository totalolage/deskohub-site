3. Update the booking form component to use react-hook-form:
```typescript
// components/BookingForm.tsx
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import { submitBooking } from "@/app/actions/booking"
import { useAction } from "next-safe-action/hooks"
import { Button } from "@/components/ui/button"
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@/components/ui/form"
import { Input } from "@/components/ui/input"
import { i } from "@/paraglide/messages"

// Define form schema using Zod (reuse existing schema from booking.ts)
const formSchema = z.object({
  date: z.string().min(1, { message: i("validation.required") }),
  time: z.string().min(1, { message: i("validation.required") }),
  guestCount: z.number().min(1).max(20),
  name: z.string().min(2),
  email: z.string().email(),
  phone: z.string().optional(),
  specialRequests: z.string().optional(),
})

export function BookingForm() {
  // Initialize react-hook-form with Zod resolver
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
    defaultValues: {
      date: "",
      time: "",
      guestCount: 2,
      name: "",
      email: "",
      phone: "",
      specialRequests: "",
    },
  })

  // Use next-safe-action hook
  const { execute, result, status } = useAction(submitBooking)

  // Handle form submission
  function onSubmit(values: z.infer<typeof formSchema>) {
    execute(values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        {/* Date field */}
        <FormField
          control={form.control}
          name="date"
          render={({ field }) => (
            <FormItem>
              <FormLabel>{i("booking.date")}</FormLabel>
              <FormControl>
                <Input type="date" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        
        {/* Implement remaining form fields with shadcn/ui components */}
        
        {/* Display server-side errors if any */}
        {result.serverError && (
          <div className="text-red-500">{result.serverError}</div>
        )}
        
        {/* Submit button with loading state */}
        <Button type="submit" disabled={status === "executing"}>
          {status === "executing" ? i("booking.submitting") : i("booking.submit")}
        </Button>
      </form>
    </Form>
  )
}
```

  phone: z.string().optional(),
  specialRequests: z.string().optional(),
});

// Create the server action
export const submitBooking = actionClient.create(
  bookingSchema,
  async (data) => {
    const locale = getCurrentLocale();
    
    try {

# Test Strategy:
1. Unit testing for form validation:
   - Create test cases for each form field validation rule
   - Test with valid inputs to ensure they pass validation
   - Test with invalid inputs to verify proper error messages are displayed
   - Verify that localized error messages are correctly generated

2. Integration testing for form submission:
   - Test the form submission process end-to-end
   - Verify that the form data is correctly passed to the server action
   - Test error handling for server-side validation failures
   - Verify success and error states are properly displayed to the user

3. Component testing:
   - Test the BookingForm component with React Testing Library
   - Verify form field rendering and interaction
   - Test form reset functionality
   - Verify form accessibility (labels, ARIA attributes, keyboard navigation)

4. User experience testing:
   - Verify that validation errors appear inline next to the relevant fields
   - Test that validation happens in real-time as users type (not just on submit)
   - Verify that the form maintains state between validation errors
   - Test the loading state during form submission

5. Accessibility testing:
   - Verify that all form elements have proper labels and ARIA attributes
   - Test keyboard navigation through the form
   - Verify that error messages are properly associated with form fields
   - Test with screen readers to ensure the form is accessible

6. Cross-browser testing:
   - Test the form in multiple browsers (Chrome, Firefox, Safari, Edge)
   - Verify responsive behavior on different screen sizes
   - Test on mobile devices to ensure touch-friendly interaction
