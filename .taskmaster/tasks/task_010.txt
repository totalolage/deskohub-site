3. Replace Arktype schema with Zod schema in booking.ts:
```typescript
// app/actions/booking.ts
import { createSafeActionClient } from 'next-safe-action';
import { z } from 'zod';
import { i } from '@/paraglide/messages';
import { getCurrentLocale } from '@/paraglide/runtime';

// Create safe action client
const actionClient = createSafeActionClient();

// Define validation schema with Zod
const bookingSchema = z.object({
  date: z.string().min(1, { message: i('validation.date.required') }),
  time: z.string().min(1, { message: i('validation.time.required') }),
  guestCount: z.number().min(1, { message: i('validation.guests.minimum') }).max(10, { message: i('validation.guests.maximum') }),
  name: z.string().min(2, { message: i('validation.name.minimum') }),
  email: z.string().email({ message: i('validation.email.invalid') }),
  phone: z.string().min(9, { message: i('validation.phone.invalid') }),
  specialRequests: z.string().optional(),
});

// Type inference from Zod schema
type BookingFormData = z.infer<typeof bookingSchema>;

// Create server action with Zod validation
export const submitBooking = actionClient.create(
  bookingSchema,
  async (data: BookingFormData) => {
    try {
      // Existing booking logic remains the same
      // ...
      
      return {
        success: true,
        message: i('booking.success')
      };
    } catch (error) {
      return {
        success: false,
        message: i('booking.error.generic')
      };
    }
  }
);
```
