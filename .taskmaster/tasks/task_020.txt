1. **Create the Menu Page Structure**:
   ```typescript
   // app/[locale]/menu/page.tsx
   import { createTranslator } from '@/paraglide/runtime';
   import { PageHeader } from '@/components/ui/page-header';
   import { MenuCategories } from '@/components/menu/menu-categories';
   import { Metadata } from 'next';
   
   // SEO metadata generation
   export async function generateMetadata({ params }): Promise<Metadata> {
     const { locale } = params;
     const t = createTranslator(locale);
     
     return {
       title: t('menu.pageTitle'),
       description: t('menu.pageDescription'),
       openGraph: {
         title: t('menu.pageTitle'),
         description: t('menu.pageDescription'),
         images: ['/images/menu-preview.jpg'],
       }
     };
   }
   
   export default function MenuPage({ params }) {
     const { locale } = params;
     const t = createTranslator(locale);
     
     return (
       <div className="container py-8 space-y-8">
         <PageHeader 
           title={t('menu.title')} 
           description={t('menu.description')} 
         />
         <MenuCategories locale={locale} />
       </div>
     );
   }
   ```

2. **Create Menu Category Components**:
   ```typescript
   // components/menu/menu-categories.tsx
   import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
   import { createTranslator } from '@/paraglide/runtime';
   import { MenuCategory } from './menu-category';
   import { getMenuItems } from '@/lib/menu';

   export function MenuCategories({ locale }) {
     const t = createTranslator(locale);
     const menuItems = getMenuItems(locale);
     
     return (
       <Tabs defaultValue="coffee" className="w-full">
         <TabsList className="mb-8">
           <TabsTrigger value="coffee">{t('menu.categories.coffee')}</TabsTrigger>
           <TabsTrigger value="tea">{t('menu.categories.tea')}</TabsTrigger>
           <TabsTrigger value="beverages">{t('menu.categories.beverages')}</TabsTrigger>
           <TabsTrigger value="snacks">{t('menu.categories.snacks')}</TabsTrigger>
           <TabsTrigger value="meals">{t('menu.categories.meals')}</TabsTrigger>
         </TabsList>
         
         <TabsContent value="coffee">
           <MenuCategory items={menuItems.coffee} />
         </TabsContent>
         <TabsContent value="tea">
           <MenuCategory items={menuItems.tea} />
         </TabsContent>
         <TabsContent value="beverages">
           <MenuCategory items={menuItems.beverages} />
         </TabsContent>
         <TabsContent value="snacks">
           <MenuCategory items={menuItems.snacks} />
         </TabsContent>
         <TabsContent value="meals">
           <MenuCategory items={menuItems.meals} />
         </TabsContent>
       </Tabs>
     );
   }
   ```

3. **Create Menu Item Component**:
   ```typescript
   // components/menu/menu-category.tsx
   import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
   import { Badge } from '@/components/ui/badge';
   import { MenuItem } from '@/types/menu';

   interface MenuCategoryProps {
     items: MenuItem[];
   }

   export function MenuCategory({ items }: MenuCategoryProps) {
     return (
       <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
         {items.map((item) => (
           <Card key={item.id}>
             <CardHeader>
               <div className="flex justify-between items-start">
                 <CardTitle>{item.name}</CardTitle>
                 <span className="font-bold">{item.price} Kč</span>
               </div>
               <CardDescription>{item.description}</CardDescription>
             </CardHeader>
             <CardContent>
               {item.allergens && (
                 <div className="flex gap-2 flex-wrap">
                   {item.allergens.map((allergen) => (
                     <Badge key={allergen} variant="outline">{allergen}</Badge>
                   ))}
                 </div>
               )}
             </CardContent>
           </Card>
         ))}
       </div>
     );
   }
   ```

# Test Strategy:
## Test Strategy

1. **Functionality Testing**:
   - Verify the Menu page loads correctly at `/menu` and `/en/menu` routes
   - Test that all category tabs work correctly and display the appropriate menu items
   - Confirm that prices, descriptions, and allergen information display correctly
   - Test that the page is fully responsive on different screen sizes (mobile, tablet, desktop)

2. **Internationalization Testing**:
   - Verify all text elements are properly translated in both Czech and English
   - Test language switching between Czech and English versions of the menu
   - Confirm that menu items display the correct language-specific descriptions
   - Verify that prices and allergen information are consistent across languages

3. **Accessibility Testing**:
   - Test keyboard navigation through the menu categories and items
   - Verify proper heading hierarchy for screen readers
   - Check that allergen information is accessible and clearly indicated
   - Test with screen readers to ensure all content is properly announced

4. **Performance Testing**:
   - Measure initial load time of the menu page
   - Test performance with a large number of menu items
   - Verify that images (if added) are properly optimized
   - Check for any layout shifts during page load

5. **SEO Testing**:
   - Verify that metadata is correctly generated for both language versions
   - Confirm that page titles and descriptions are appropriate for search engines
   - Check that structured data (if implemented) is valid

6. **Visual Regression Testing**:
   - Compare the implemented design against mockups or design specifications
   - Verify consistent styling with the rest of the application
   - Test dark/light mode appearance if the site supports theme switching

7. **Integration Testing**:
   - Verify navigation to and from the menu page works correctly
   - Test that the menu page integrates properly with the site's header and footer
   - Check that any links within the menu page function correctly
