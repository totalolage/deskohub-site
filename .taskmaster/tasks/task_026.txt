# Task ID: 26
# Title: Implement Safe Area Insets for Device Screen Cutouts
# Status: pending
# Dependencies: 1, 6, 23
# Priority: medium
# Description: Implement safe inset areas throughout the application to properly handle devices with notches, rounded corners, or other screen cutouts, ensuring UI elements remain visible and accessible on all device types.
# Details:
## Objective
Ensure all UI components respect safe area insets on devices with notches, rounded corners, or other screen cutouts (particularly iOS devices, but also applicable to some Android devices).

## Implementation Steps

1. **Create a Safe Area Context Provider**:
```tsx
// components/providers/safe-area-provider.tsx
import React, { createContext, useContext, useEffect, useState } from 'react';

interface SafeAreaInsets {
  top: number;
  right: number;
  bottom: number;
  left: number;
}

const SafeAreaContext = createContext<SafeAreaInsets>({
  top: 0,
  right: 0,
  bottom: 0,
  left: 0,
});

export const useSafeArea = () => useContext(SafeAreaContext);

export function SafeAreaProvider({ children }: { children: React.ReactNode }) {
  const [insets, setInsets] = useState<SafeAreaInsets>({
    top: 0,
    right: 0,
    bottom: 0,
    left: 0,
  });

  useEffect(() => {
    // For iOS devices using env() CSS variables
    const computeInsets = () => {
      if (typeof window !== 'undefined' && window.CSS && CSS.supports('padding-top: env(safe-area-inset-top)')) {
        // Get computed values from a temporary element
        const tempEl = document.createElement('div');
        tempEl.style.paddingTop = 'env(safe-area-inset-top)';
        tempEl.style.paddingRight = 'env(safe-area-inset-right)';
        tempEl.style.paddingBottom = 'env(safe-area-inset-bottom)';
        tempEl.style.paddingLeft = 'env(safe-area-inset-left)';
        document.body.appendChild(tempEl);
        
        const computedStyle = window.getComputedStyle(tempEl);
        setInsets({
          top: parseInt(computedStyle.paddingTop) || 0,
          right: parseInt(computedStyle.paddingRight) || 0,
          bottom: parseInt(computedStyle.paddingBottom) || 0,
          left: parseInt(computedStyle.paddingLeft) || 0,
        });
        
        document.body.removeChild(tempEl);
      }
    };

    computeInsets();
    window.addEventListener('resize', computeInsets);
    
    return () => {
      window.removeEventListener('resize', computeInsets);
    };
  }, []);

  return (
    <SafeAreaContext.Provider value={insets}>
      {children}
    </SafeAreaContext.Provider>
  );
}
```

5. **Create a Custom Hook for Tailwind Classes**:
```tsx
// hooks/use-safe-area-classes.ts
import { useSafeArea } from '@/components/providers/safe-area-provider';
import { cn } from '@/lib/utils';

export function useSafeAreaClasses() {
  const insets = useSafeArea();
  
  return {
    top: (className?: string) => cn(insets.top > 0 ? 'safe-padding-top' : '', className),
    right: (className?: string) => cn(insets.right > 0 ? 'safe-padding-right' : '', className),
    bottom: (className?: string) => cn(insets.bottom > 0 ? 'safe-padding-bottom' : '', className),
    left: (className?: string) => cn(insets.left > 0 ? 'safe-padding-left' : '', className),
    horizontal: (className?: string) => cn(
      (insets.left > 0 || insets.right > 0) ? 'safe-padding-horizontal' : '', 
      className
    ),
    vertical: (className?: string) => cn(
      (insets.top > 0 || insets.bottom > 0) ? 'safe-padding-vertical' : '', 
      className
    ),
    all: (className?: string) => cn(
      (insets.top > 0 || insets.right > 0 || insets.bottom > 0 || insets.left > 0) ? 'safe-padding' : '',
      className
    ),
  };
}
```

9. **Create a Safe Area Debug Mode**:
```tsx
// components/providers/safe-area-provider.tsx
// Add to the provider:
const [debugMode, setDebugMode] = useState(false);

useEffect(() => {
  // Toggle debug with keyboard shortcut (e.g., Alt+Shift+D)
  const handleKeyDown = (e: KeyboardEvent) => {
    if (e.altKey && e.shiftKey && e.key === 'D') {
      setDebugMode(prev => !prev);
    }
  };
  
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, []);

// Return with debug visualization
return (
  <SafeAreaContext.Provider value={insets}>
    {children}
    {debugMode && (
      <>
        <div className="fixed top-0 left-0 right-0 bg-red-500 opacity-50" style={{ height: `${insets.top}px` }} />
        <div className="fixed bottom-0 left-0 right-0 bg-red-500 opacity-50" style={{ height: `${insets.bottom}px` }} />
        <div className="fixed top-0 left-0 bottom-0 bg-red-500 opacity-50" style={{ width: `${insets.left}px` }} />
        <div className="fixed top-0 right-0 bottom-0 bg-red-500 opacity-50" style={{ width: `${insets.right}px` }} />
      </>
    )}
  </SafeAreaContext.Provider>
);
```

# Test Strategy:
## Test Strategy

1. **Device Testing Matrix**:
   - Test on iOS devices with notches (iPhone X and newer)
   - Test on iOS devices with Dynamic Island (iPhone 14 Pro and newer)
   - Test on Android devices with notches or punch-hole cameras
   - Test on devices with rounded corners
   - Test on standard devices without special cutouts (as control)

2. **Orientation Testing**:
   - Test in portrait mode
   - Test in landscape mode (both left and right orientations)
   - Verify insets adjust correctly when rotating the device

3. **Component-Specific Testing**:
   - **Header/Navigation**: Verify content doesn't overlap with notches or status bar
   - **Mobile Menu**: Check that the slide-out menu respects safe areas
   - **Fixed Position Elements**: Ensure bottom navigation bars, FABs, and other fixed elements respect safe areas
   - **Full-Screen Modals**: Test that modals and dialogs extend properly but keep content within safe areas
   - **Gallery/Lightbox**: Verify full-screen images respect safe areas
   - **Forms**: Check that input fields and buttons near screen edges are fully accessible

4. **Visual Regression Testing**:
   - Take screenshots before and after implementation on various devices
   - Compare to ensure no unintended layout changes occurred
   - Verify UI elements maintain proper spacing from screen edges

5. **Debug Mode Testing**:
   - Verify the debug visualization correctly shows safe area insets
   - Confirm the keyboard shortcut (Alt+Shift+D) correctly toggles debug mode
   - Check that debug visualization accurately reflects actual insets

6. **Browser Compatibility**:
   - Test in Safari on iOS (primary target)
   - Test in Chrome on Android
   - Test in other mobile browsers (Firefox, Samsung Internet)
   - Verify fallback behavior works correctly on unsupported browsers

7. **Performance Testing**:
   - Measure any impact on rendering performance
   - Check for layout shifts when insets are applied
   - Verify smooth transitions when device orientation changes

8. **Accessibility Testing**:
   - Verify all interactive elements remain fully accessible
   - Check that touch targets near screen edges are not compromised
   - Ensure no content is hidden or difficult to access due to insets

9. **Manual Testing Checklist**:
   - [ ] All fixed headers respect top safe area inset
   - [ ] All fixed footers respect bottom safe area inset
   - [ ] Side menus respect left/right safe areas
   - [ ] Full-width containers properly handle horizontal safe areas
   - [ ] No content is cut off by notches or rounded corners
   - [ ] No interactive elements are positioned under notches or rounded corners
   - [ ] Layout adjusts properly when rotating device
   - [ ] Safe areas are applied consistently across all pages and components
