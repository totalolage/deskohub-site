# Details:
1. Create a Next.js middleware for language routing:
```javascript
// middleware.js
import { NextResponse } from 'next/server';
import { availableLanguageTags } from '@/paraglide/runtime';

// Define the default language (Czech)
const defaultLanguage = 'cs';

// Function to get language from pathname
function getLanguageFromPathname(pathname) {
  // Check if the first segment matches a supported language
  const firstSegment = pathname.split('/')[1];
  return availableLanguageTags.includes(firstSegment) ? firstSegment : defaultLanguage;
}

export function middleware(request) {
  const { pathname } = request.nextUrl;
  
  // Skip for assets, api routes, etc.
  if (
    pathname.startsWith('/_next') ||
    pathname.startsWith('/api') ||
    pathname.includes('.') // files like favicon.ico
  ) {
    return NextResponse.next();
  }
  
  // Get language from URL
  const language = getLanguageFromPathname(pathname);
  
  // Set language in request headers for server components
  const requestHeaders = new Headers(request.headers);
  requestHeaders.set('x-language', language);
  
  return NextResponse.next({
    request: {
      headers: requestHeaders,
    },
  });
}

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico).*)'],
};
```

2. Update the language switcher component to use proper Next.js navigation:
```jsx
// src/components/LanguageSwitcher.jsx
import Link from 'next/link';
import { usePathname } from 'next/navigation';
import { useLanguage, availableLanguageTags } from '@/paraglide/runtime';

export function LanguageSwitcher() {
  const currentLanguage = useLanguage();
  const pathname = usePathname();
  
  // Function to get the equivalent path in another language
  function getLanguagePath(targetLanguage) {
    // Remove the language prefix from current path if it exists
    let pathWithoutLanguage = pathname;
    
    // Check if current path starts with a language tag
    for (const lang of availableLanguageTags) {
      if (pathname.startsWith(`/${lang}/`)) {
        pathWithoutLanguage = pathname.substring(lang.length + 1);
        break;
      }
    }
    
    // For default language (Czech), use root path
    if (targetLanguage === 'cs') {
      return pathWithoutLanguage || '/';
    }
    
    // For other languages, prefix with language code
    return `/${targetLanguage}${pathWithoutLanguage}`;
  }
  
  return (
    <div className="language-switcher">
      {availableLanguageTags.map((lang) => (
        <Link 
          key={lang}
          href={getLanguagePath(lang)}
          className={currentLanguage === lang ? 'active' : ''}
        >
          {lang.toUpperCase()}
        </Link>
      ))}
    </div>
  );
}
```

3. Update the root layout to detect and set the language based on URL:

```jsx
// app/layout.jsx
import { headers } from 'next/headers';
import { LanguageProvider } from '@/paraglide/runtime';

export default function RootLayout({ children }) {
  // Get language from request headers (set by middleware)
  const headersList = headers();
  const language = headersList.get('x-language') || 'cs';
  
  return (
    <html lang={language}>
      <LanguageProvider language={language}>
        <body>
          {/* Your app content */}
          {children}
        </body>
      </LanguageProvider>
    </html>
  );
}
```
