4. Create a BookingPageContent component that uses the existing BookingForm:
```typescript
// src/components/booking/BookingPageContent.tsx
'use client';

import { useState } from 'react';
import { useAction } from 'next-safe-action/hooks';
import { submitBooking } from '@/app/actions/booking';
import BookingForm from '@/components/BookingForm';
import { getTranslations } from '@/paraglide/runtime';
import { Alert, AlertTitle, AlertDescription } from '@/components/ui/alert';

export default function BookingPageContent() {
  const t = getTranslations();
  const [formState, setFormState] = useState({
    date: '',
    time: '',
    guestCount: 1,
    name: '',
    email: '',
    phone: '',
    tablePreference: '',
    specialRequests: '',
    preOrders: ''
  });
  
  const { execute, result, status } = useAction(submitBooking);
  
  const handleSubmit = (formData) => {
    execute(formData);
  };
  
  const handleChange = (field, value) => {
    setFormState(prev => ({ ...prev, [field]: value }));
  };
  
  return (
    <div>
      {status === 'hasSucceeded' && result.data.success && (
        <Alert className="mb-6 bg-green-50 border-green-200">
          <AlertTitle>{t('booking.successTitle')}</AlertTitle>
          <AlertDescription>{t('booking.successMessage')}</AlertDescription>
        </Alert>
      )}
      
      {status === 'hasErrored' || (status === 'hasSucceeded' && !result.data.success) ? (
        <Alert className="mb-6 bg-red-50 border-red-200">
          <AlertTitle>{t('booking.errorTitle')}</AlertTitle>
          <AlertDescription>{t('booking.errorMessage')}</AlertDescription>
        </Alert>
      ) : null}
      
      <BookingForm 
        formState={formState}
        onChange={handleChange}
        onSubmit={handleSubmit}
        isSubmitting={status === 'isExecuting'}
      />
    </div>
  );
}
```
