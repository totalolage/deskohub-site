2. Create a utility function to detect the current locale in server actions:
```typescript
// lib/locale-utils.ts
import { cookies } from 'next/headers';
import { availableLanguageTags, sourceLanguageTag } from '@/paraglide/runtime';

export function getCurrentLocale(): string {
  const cookieStore = cookies();
  const localeCookie = cookieStore.get('NEXT_LOCALE')?.value;
  
  if (localeCookie && availableLanguageTags.includes(localeCookie)) {
    return localeCookie;
  }
  
  return sourceLanguageTag;
}
```

3. Update the Zod error map to use Paraglide messages:
```typescript
// lib/zod-i18n.ts
import { z } from 'zod';
import { m } from '@/paraglide/messages';
import { getCurrentLocale } from './locale-utils';

export function setupZodI18n() {
  // Set the error map to use Paraglide translations
  z.setErrorMap((issue, ctx) => {
    const locale = getCurrentLocale();
    
    let message: string;
    switch (issue.code) {
      case z.ZodIssueCode.invalid_type:
        if (issue.received === 'undefined' || issue.received === 'null') {
          message = m.validation.required({ locale });
        } else {
          message = `Expected ${issue.expected}, received ${issue.received}`;
        }
        break;
      case z.ZodIssueCode.invalid_string:
        if (issue.validation === 'email') {
          message = m.validation.email({ locale });
        } else if (issue.validation === 'min') {
          message = m.validation.minLength({ locale, min: issue.minimum });
        } else if (issue.validation === 'max') {
          message = m.validation.maxLength({ locale, max: issue.maximum });
        } else {
          message = ctx.defaultError;
        }
        break;
      case z.ZodIssueCode.too_small:
        if (issue.type === 'number') {
          message = m.validation.min({ locale, min: issue.minimum });
        } else {
          message = m.validation.minLength({ locale, min: issue.minimum });
        }
        break;
      case z.ZodIssueCode.too_big:
        if (issue.type === 'number') {
          message = m.validation.max({ locale, max: issue.maximum });
        } else {
          message = m.validation.maxLength({ locale, max: issue.maximum });
        }
        break;
      // Add other cases as needed
      default:
        message = ctx.defaultError;
    }

    return { message };
  });
}
```

    })
