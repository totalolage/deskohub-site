2. **Create Validation Utility Functions**:
   ```typescript
   // lib/validation/business-hours.ts
   import { defaultBusinessHours, specialDays, type DayOfWeek } from '../config/business-hours';
   
   export function isWithinBusinessHours(date: Date): boolean {
     const dayOfWeek = date.toLocaleLowerCase('en-US') as DayOfWeek;
     const timeString = date.toTimeString().substring(0, 5); // Get HH:MM format
     
     // Check if it's a special day
     const dateString = date.toISOString().split('T')[0]; // YYYY-MM-DD
     const specialDay = specialDays.find(day => day.date === dateString);
     
     if (specialDay) {
       if (specialDay.closed) return false;
       if (specialDay.hours) {
         return timeString >= specialDay.hours.open && timeString <= specialDay.hours.close;
       }
     }
     
     // Regular business hours check
     const hours = defaultBusinessHours[dayOfWeek];
     return timeString >= hours.open && timeString <= hours.close;
   }
   
   export function getNextAvailableTime(date: Date): Date {
     // If outside business hours, return the next opening time
     const dayOfWeek = date.toLocaleLowerCase('en-US') as DayOfWeek;
     const dateString = date.toISOString().split('T')[0];
     const specialDay = specialDays.find(day => day.date === dateString);
     
     // Handle special days
     if (specialDay) {
       if (specialDay.closed) {
         // Return next day's opening time
         const nextDay = new Date(date);
         nextDay.setDate(nextDay.getDate() + 1);
         return getNextAvailableTime(nextDay);
       }
       
       if (specialDay.hours) {
         const [openHour, openMinute] = specialDay.hours.open.split(':').map(Number);
         const timeString = date.toTimeString().substring(0, 5);
         
         if (timeString < specialDay.hours.open) {
           // Before opening time, return opening time
           const openTime = new Date(date);
           openTime.setHours(openHour, openMinute, 0, 0);
           return openTime;
         }
         
         if (timeString > specialDay.hours.close) {
           // After closing time, return next day's opening time
           const nextDay = new Date(date);
           nextDay.setDate(nextDay.getDate() + 1);
           return getNextAvailableTime(nextDay);
         }
       }
     }
     
     // Regular business hours handling
     const hours = defaultBusinessHours[dayOfWeek];
     const [openHour, openMinute] = hours.open.split(':').map(Number);
     const [closeHour, closeMinute] = hours.close.split(':').map(Number);
     
     const timeString = date.toTimeString().substring(0, 5);
     
     if (timeString < hours.open) {
       // Before opening time, return opening time
       const openTime = new Date(date);
       openTime.setHours(openHour, openMinute, 0, 0);
       return openTime;
     }
     
     if (timeString > hours.close) {
       // After closing time, return next day's opening time
       const nextDay = new Date(date);
       nextDay.setDate(nextDay.getDate() + 1);
       return getNextAvailableTime(nextDay);
     }
     
     return date; // Already within business hours
   }
   
   export function getBusinessHoursForDate(date: Date): { open: string; close: string } | null {
     const dayOfWeek = date.toLocaleLowerCase('en-US') as DayOfWeek;
     const dateString = date.toISOString().split('T')[0];
     
     // Check for special days
     const specialDay = specialDays.find(day => day.date === dateString);
     if (specialDay) {
       if (specialDay.closed) return null;
       if (specialDay.hours) return specialDay.hours;
     }
     
     // Return regular business hours
     return defaultBusinessHours[dayOfWeek];
   }
   ```

3. **Update Zod Validation Schema**:
   ```typescript
   // lib/validation/reservation-schema.ts
   import { z } from 'zod';
   import { isWithinBusinessHours, getBusinessHoursForDate } from './business-hours';
   import { createTranslator } from '@/paraglide/runtime';
   
   export const createReservationSchema = (locale: string) => {
     const t = createTranslator(locale);
     
     return z.object({
       date: z.date({
         required_error: t('validation.required'),
         invalid_type_error: t('validation.invalidDate'),
       }),
       time: z.string({
         required_error: t('validation.required'),
       }).refine((time) => {
         // Basic time format validation
         return /^([01]?[0-9]|2[0-3]):[0-5][0-9]$/.test(time);
       }, {
         message: t('validation.invalidTime'),
       }),
       // ... other fields
     }).refine((data) => {
       // Combine date and time for business hours validation
       const reservationDateTime = new Date(data.date);
       const [hours, minutes] = data.time.split(':').map(Number);
       reservationDateTime.setHours(hours, minutes, 0, 0);
       
       return isWithinBusinessHours(reservationDateTime);
     }, {
       message: (data) => {
         const date = new Date(data.date);
         const hours = getBusinessHoursForDate(date);
         
         if (!hours) {
           return t('validation.closedDay');
         }
         
         return t('validation.outsideBusinessHours', { 
           open: hours.open, 
           close: hours.close 
         });
       },
       path: ['time'], // Show error on the time field
     });
   };
   ```

4. **Update Client-Side Form Validation**:
   ```tsx
   // components/reservation/reservation-form.tsx
   import { useState } from 'react';
   import { useForm } from 'react-hook-form';
   import { zodResolver } from '@hookform/resolvers/zod';
   import { createReservationSchema } from '@/lib/validation/reservation-schema';
   import { createTranslator } from '@/paraglide/runtime';
   import { getBusinessHoursForDate } from '@/lib/validation/business-hours';
   
   export function ReservationForm({ locale }) {
     const t = createTranslator(locale);
     const schema = createReservationSchema(locale);
     
     const {
       register,
       handleSubmit,
       formState: { errors },
       watch,
       setValue,
     } = useForm({
       resolver: zodResolver(schema),
     });
     
     const selectedDate = watch('date');
     const businessHours = selectedDate ? getBusinessHoursForDate(new Date(selectedDate)) : null;
     
     return (
       <form onSubmit={handleSubmit(onSubmit)}>
         {/* Date field */}
         <div className="form-group">
           <label htmlFor="date">{t('reservation.date')}</label>
           <input
             type="date"
             id="date"
             {...register('date')}
           />
           {errors.date && (
             <p className="error">{errors.date.message}</p>
           )}
         </div>
         
         {/* Time field */}
         <div className="form-group">
           <label htmlFor="time">{t('reservation.time')}</label>
           <input
             type="time"
             id="time"
             {...register('time')}
           />
           {errors.time && (
             <p className="error">{errors.time.message}</p>
           )}
           
           {/* Display business hours information */}
           {businessHours ? (
             <p className="text-sm text-muted-foreground">
               {t('reservation.businessHours', { 
                 open: businessHours.open, 
                 close: businessHours.close 
               })}
             </p>
           ) : selectedDate ? (
             <p className="text-sm text-destructive">
               {t('reservation.closedDay')}
             </p>
           ) : null}
         </div>
         
         {/* Other form fields */}
         
         <button type="submit">{t('reservation.submit')}</button>
       </form>
     );
   }
   ```

   // app/api/reservations/route.ts
   import { NextRequest, NextResponse } from 'next/server';
   import { createReservationSchema } from '@/lib/validation/reservation-schema';
   import { isWithinBusinessHours } from '@/lib/validation/business-hours';
   
   export async function POST(request: NextRequest) {
     try {
       const body = await request.json();
       const locale = request.headers.get('x-locale') || 'cs';
       
       // Validate the request body
       const schema = createReservationSchema(locale);
       const result = schema.safeParse(body);
       
       if (!result.success) {
         return NextResponse.json(
           { error: result.error.format() },
           { status: 400 }
         );
       }
       
       // Additional server-side validation
       const reservationDateTime = new Date(body.date);
       const [hours, minutes] = body.time.split(':').map(Number);
       reservationDateTime.setHours(hours, minutes, 0, 0);
       
       if (!isWithinBusinessHours(reservationDateTime)) {
         return NextResponse.json(
           { error: 'Reservation time is outside business hours' },
           { status: 400 }
         );
       }
       
       // Process the reservation
       // ...
       
       return NextResponse.json({ success: true });
     } catch (error) {
       return NextResponse.json(
         { error: 'Internal server error' },
         { status: 500 }
       );
     }
   }

   // tests/lib/validation/business-hours.test.ts
   import { describe, it, expect } from 'vitest';
   import { 
     isWithinBusinessHours,

     getBusinessHoursForDate 
   } from '@/lib/validation/business-hours';
   
   describe('Business Hours Validation', () => {
     it('should correctly identify times within business hours', () => {
       // Regular weekday test
       const weekdayInHours = new Date('2024-06-10T14:30:00'); // Monday at 2:30 PM
       expect(isWithinBusinessHours(weekdayInHours)).toBe(true);
       
       // Regular weekend test
       const weekendInHours = new Date('2024-06-15T10:30:00'); // Saturday at 10:30 AM
       expect(isWithinBusinessHours(weekendInHours)).toBe(true);
     });
     
     it('should correctly identify times outside business hours', () => {
       // Before opening
       const beforeOpening = new Date('2024-06-10T07:30:00'); // Monday at 7:30 AM
       expect(isWithinBusinessHours(beforeOpening)).toBe(false);
       
       // After closing
       const afterClosing = new Date('2024-06-10T22:30:00'); // Monday at 10:30 PM
       expect(isWithinBusinessHours(afterClosing)).toBe(false);
     });
     
     it('should handle special days correctly', () => {
       // Closed holiday
       const christmasDay = new Date('2024-12-25T12:00:00');
       expect(isWithinBusinessHours(christmasDay)).toBe(false);
       
       // Special hours day
       const newYearsEve = new Date('2024-12-31T17:00:00'); // 5 PM, within special hours
       expect(isWithinBusinessHours(newYearsEve)).toBe(true);
       
       const newYearsEveLate = new Date('2024-12-31T19:00:00'); // 7 PM, outside special hours
       expect(isWithinBusinessHours(newYearsEveLate)).toBe(false);
     });
     
     it('should correctly calculate next available time', () => {
       // Before opening hours
       const beforeOpening = new Date('2024-06-10T07:30:00'); // Monday at 7:30 AM
       const expectedOpeningTime = new Date('2024-06-10T08:00:00');
       expect(getNextAvailableTime(beforeOpening)).toEqual(expectedOpeningTime);
       
       // After closing hours
       const afterClosing = new Date('2024-06-10T22:30:00'); // Monday at 10:30 PM
       const nextDayOpening = new Date('2024-06-11T08:00:00'); // Tuesday at 8:00 AM
       expect(getNextAvailableTime(afterClosing)).toEqual(nextDayOpening);
       
       // On a closed day
       const christmasDay = new Date('2024-12-25T12:00:00');
       const dayAfterChristmas = new Date('2024-12-26T08:00:00');
       expect(getNextAvailableTime(christmasDay)).toEqual(dayAfterChristmas);
     });
     
     it('should return correct business hours for a date', () => {
       // Regular weekday
       const weekday = new Date('2024-06-10'); // Monday
       expect(getBusinessHoursForDate(weekday)).toEqual({ open: '08:00', close: '22:00' });
