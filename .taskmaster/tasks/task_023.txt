1. **Create the Mobile Menu Component**:
```tsx
// components/ui/mobile-menu.tsx
import { useState, useEffect } from 'react';
import { createTranslator } from '@/paraglide/runtime';
import { Button } from '@/components/ui/button';
import { Sheet, SheetContent, SheetTrigger } from '@/components/ui/sheet';
import { Menu } from 'lucide-react';
import { NavigationLinks } from '@/components/navigation-links';
import { cn } from '@/lib/utils';

interface MobileMenuProps {
  locale: string;
}

export function MobileMenu({ locale }: MobileMenuProps) {
  const t = createTranslator(locale);
  const [isOpen, setIsOpen] = useState(false);
  
  // Close menu when screen size changes to desktop
  useEffect(() => {
    const handleResize = () => {
      if (window.innerWidth >= 768) {
        setIsOpen(false);
      }
    };
    
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return (
    <Sheet open={isOpen} onOpenChange={setIsOpen}>
      <SheetTrigger asChild>
        <Button 
          variant="ghost" 
          size="icon" 
          className="md:hidden" 
          aria-label={t('common.mobileMenu')}
        >
          <Menu className="h-6 w-6" />
        </Button>
      </SheetTrigger>
      <SheetContent side="right" className="p-0">
        <nav className="flex flex-col h-full p-6">
          <NavigationLinks 
            locale={locale} 
            className="flex flex-col space-y-4" 
            itemClassName="text-lg py-2"
            onClick={() => setIsOpen(false)}
          />
        </nav>
      </SheetContent>
    </Sheet>
  );
}
```

3. **Update the Navigation Links Component**:
```tsx
// components/navigation-links.tsx
import Link from 'next/link';
import { usePathname } from 'next/navigation';
import { createTranslator } from '@/paraglide/runtime';
import { cn } from '@/lib/utils';

interface NavigationLinksProps {
  locale: string;
  className?: string;
  itemClassName?: string;
  onClick?: () => void;
}

export function NavigationLinks({ 
  locale, 
  className, 
  itemClassName,
  onClick 
}: NavigationLinksProps) {
  const t = createTranslator(locale);
  const pathname = usePathname();
  
  // Define navigation items with their paths and labels
  const navItems = [
    { path: `/${locale === 'en' ? 'en/' : ''}`, label: t('nav.home') },
    { path: `/${locale === 'en' ? 'en/' : ''}desks`, label: t('nav.desks') },
    { path: `/${locale === 'en' ? 'en/' : ''}gallery`, label: t('nav.gallery') },
    { path: `/${locale === 'en' ? 'en/' : ''}menu`, label: t('nav.menu') },
    { path: `/${locale === 'en' ? 'en/' : ''}training-room`, label: t('nav.trainingRoom') },
    { path: `/${locale === 'en' ? 'en/' : ''}contact`, label: t('nav.contact') },
  ];
  
  return (
    <ul className={cn('list-none', className)}>
      {navItems.map((item) => {
        const isActive = pathname === item.path || 
                        (item.path !== `/${locale === 'en' ? 'en/' : ''}` && 
                         pathname.startsWith(item.path));
        
        return (
          <li key={item.path}>
            <Link 
              href={item.path}
              className={cn(
                'transition-colors hover:text-primary',
                isActive ? 'font-medium text-primary' : 'text-muted-foreground',
                itemClassName
              )}
              onClick={onClick}
            >
              {item.label}
            </Link>
          </li>
        );
      })}
    </ul>
  );
}
```

```tsx
// Enhanced accessibility in mobile-menu.tsx
useEffect(() => {
  // Trap focus within the menu when open
  const handleKeyDown = (e: KeyboardEvent) => {
    if (isOpen && e.key === 'Escape') {
      setIsOpen(false);
    }
  };
  
  document.addEventListener('keydown', handleKeyDown);
  return () => document.removeEventListener('keydown', handleKeyDown);
}, [isOpen, setIsOpen]);

// Add aria-expanded attribute to the button
<Button 
  variant="ghost" 
  size="icon" 
  className="md:hidden" 
  aria-label={t('common.mobileMenu')}
  aria-expanded={isOpen}
>
  <Menu className="h-6 w-6" />
</Button>
```

# Test Strategy:
## Test Strategy

1. **Responsive Layout Testing**:
   - Test on various screen sizes (320px, 375px, 414px, 768px, 1024px, 1440px)
   - Verify the mobile menu is hidden on screens ≥768px (md breakpoint)
   - Verify the desktop navigation is hidden on screens <768px
   - Test the transition between mobile and desktop layouts by resizing the browser window
   - Ensure the mobile menu closes automatically when the screen is resized to desktop width

2. **Functionality Testing**:
   - Verify the hamburger menu button appears correctly on mobile screens
   - Test that clicking the hamburger icon opens the mobile menu
   - Confirm all navigation links are visible and clickable in the mobile menu
   - Verify that clicking a navigation link correctly navigates to the intended page
   - Confirm the mobile menu closes after selecting a navigation link
   - Test that the close button (if implemented) properly dismisses the menu

3. **Animation Testing**:
   - Verify smooth opening/closing animations for the mobile menu
   - Test that animations complete without visual glitches
   - Ensure animations work consistently across different browsers and devices
   - Verify animations respect user preferences for reduced motion

4. **Touch Interaction Testing**:
   - Test touch gestures (tap, swipe) on actual mobile devices
   - Verify swipe-to-close functionality works as expected
   - Ensure touch targets are appropriately sized (at least 44×44px)
   - Test that scrolling within the menu works if content exceeds screen height

5. **Accessibility Testing**:
   - Verify the menu can be operated using keyboard navigation only
   - Test with screen readers (VoiceOver, NVDA) to ensure proper announcements
   - Verify proper focus management when opening/closing the menu
   - Check that all interactive elements have appropriate ARIA attributes
   - Verify sufficient color contrast for all text and UI elements
   - Test with browser zoom up to 200%

6. **Internationalization Testing**:
   - Verify the menu works correctly in both Czech and English
   - Test that translated strings display properly in the mobile menu
   - Ensure right-to-left language support works if applicable
   - Check that text doesn't overflow containers when translated

7. **Cross-browser and Cross-device Testing**:
   - Test on iOS Safari, Chrome for Android, and other mobile browsers
   - Verify functionality on older devices with smaller screens
   - Test on tablets in both portrait and landscape orientations
