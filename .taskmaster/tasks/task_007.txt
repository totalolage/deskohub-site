2. Create navigation helper functions:
```typescript
// src/utils/navigation.ts
import { useRouter } from 'next/navigation';
import { AppRoutes, RouteParams, QueryParams } from '@/types/routes';

// Type-safe navigation function
export function createTypedLink<T extends keyof AppRoutes>(
  route: T,
  options?: AppRoutes[T]
): string {
  let path = route as string;
  
  // Replace dynamic parameters
  if (options?.params) {
    Object.entries(options.params).forEach(([key, value]) => {
      path = path.replace(`[${key}]`, encodeURIComponent(String(value)));
    });
  }
  
  // Add query parameters
  if (options?.query) {
    const queryString = Object.entries(options.query)
      .filter(([_, value]) => value !== undefined)
      .map(([key, value]) => `${encodeURIComponent(key)}=${encodeURIComponent(String(value))}`)
      .join('&');
      
    if (queryString) {
      path += `?${queryString}`;
    }
  }
  
  return path;
}

// Hook for type-safe navigation
export function useTypedRouter() {
  const router = useRouter();
  
  return {
    push: <T extends keyof AppRoutes>(route: T, options?: AppRoutes[T]) => {
      router.push(createTypedLink(route, options));
    },
    replace: <T extends keyof AppRoutes>(route: T, options?: AppRoutes[T]) => {
      router.replace(createTypedLink(route, options));
    },
    // Add other router methods as needed
  };
}
```

3. Create Link component wrapper for type safety:
```typescript
// src/components/TypedLink.tsx
import Link from 'next/link';
import { createTypedLink } from '@/utils/navigation';
import { AppRoutes } from '@/types/routes';

type TypedLinkProps<T extends keyof AppRoutes> = {
  route: T;
  options?: AppRoutes[T];
  children: React.ReactNode;
} & Omit<React.ComponentProps<typeof Link>, 'href'>;

export function TypedLink<T extends keyof AppRoutes>({
  route,
  options,
  children,
  ...props
}: TypedLinkProps<T>) {
  const href = createTypedLink(route, options);
  
  return (
    <Link href={href} {...props}>
      {children}
    </Link>
  );
}
```
